     using { anubhav.claude as common } from '../db/common';

     @path: '/catalog'

     service CatalogService {
       entity Travellers          as projection on db.Travellers;
       @readonly entity TravellerTypes   as projection on common.TravellerTypes;
       entity TravelledLocations  as projection on db.TravelledLocations;
       entity Destinations        as projection on db.Destinations;
       @readonly entity TravellerStatus  as projection on
     common.TravellerStatus;
       @readonly entity AddressTypes     as projection on common.AddressTypes;
     }
     Authorization annotations (inline or in separate section):
     annotate CatalogService.Travellers with @restrict: [
       { grant: ['READ', 'UPDATE'], to: 'Traveller', where: 'userID = $user' },
       { grant: '*', to: 'Admin' }
     ];
     annotate CatalogService.TravelledLocations with @restrict: [
       { grant: ['READ', 'CREATE', 'DELETE'], to: 'Traveller', where:
     'traveller.userID = $user' },
       { grant: '*', to: 'Admin' }

     annotate CatalogService.Destinations with @restrict: [
       { grant: ['READ', 'CREATE'], to: 'authenticated-user' }

     2.2 srv/user-management.cds

     using { anubhav.claude as db } from '../db/schema';
     using { anubhav.claude as common } from '../db/common';

     @path: '/user-management'
     @requires: 'Admin'
     service UserManagement {
       entity Users     as projection on db.Users;
       @readonly entity Roles as projection on common.Roles;
       entity UserRoles as projection on db.UserRoles;

       function lockUser(userId: String)    returns Boolean;
       function unlockUser(userId: String)  returns Boolean;
       action createUser(firstName: String, lastName: String, email: String,
     roleId: String) returns Users;
       action resetPassword(userId: String, newPassword: String) returns
     Boolean;
       action assignRole(userId: String, roleId: String)   returns Boolean;
       action unassignRole(userId: String, roleId: String) returns Boolean;
     }

     Phase 3 вҖ” Handler Files

     3.1 srv/cat-service.js (entry point вҖ” registers sub-handlers)

     const travellersHandler = require('./handlers/travellers.handler');

     3.2 srv/handlers/travellers.handler.js

     - before('CREATE', Travellers):
       - Validate firstName and lastName are non-empty вҶ’ req.error(400, 'First 
     name and last name are required')
       - Check for duplicate email вҶ’ req.error(409, 'Email already registered')
       - Auto-assign userID from req.user.id if not set
     - before('UPDATE', Travellers):
       - Validate firstName and lastName not empty if present in payload
     - after('READ', Travellers):
       - Compute fullName = (firstName + ' ' + lastName).trim() for each record
     (handle array + single)

     3.3 srv/handlers/travelled-locations.handler.js

     - before('CREATE', TravelledLocations):
       - Validate travelFrom <= travelTo вҶ’ req.error(400, 'Travel start date 
     must be before end date')
       - Query existing TravelledLocations for same traveller_ID and check date
     overlap вҶ’ req.error(409, 'Traveller already has a trip during this date 
     range')

     3.4 srv/handlers/user-management.handler.js

     Registers on the UserManagement service:
     - lockUser: find user by ID; if not found вҶ’ 404; if already locked вҶ’ return
      false; else set isLocked=true, return true
     - unlockUser: find user by ID вҶ’ 404 if missing; set isLocked=false, return
     true
     - createUser: check duplicate loginName вҶ’ req.error(409, 'User already 
     exists'); INSERT new user; if roleId valid, INSERT UserRoles record; return
      created user
     - resetPassword: validate non-empty вҶ’ 400; validate length >= 8 вҶ’ 400;
     update record
     - assignRole: validate roleId exists in Roles вҶ’ req.error(404, 'Role not 
     found'); insert UserRoles record; return true
     - unassignRole: find existing UserRoles record вҶ’ req.error(404, 'Role 
     assignment not found'); delete record; return true

     UserManagement entry point file: srv/user-management.js:
     const umHandler = require('./handlers/user-management.handler');
     module.exports = (srv) => umHandler(srv);
